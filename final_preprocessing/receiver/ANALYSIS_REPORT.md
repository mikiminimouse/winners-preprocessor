```python
import argparse
import sys
from pathlib import Path

# –î–æ–±–∞–≤–∏—Ç—å –≤ PYTHONPATH
sys.path.insert(0, str(Path(__file__).parent.parent))

from receiver.sync_db.enhanced_service import EnhancedSyncService
from receiver.downloader.enhanced_service import EnhancedProtocolDownloader
from receiver.sync_db.health_checks import run_comprehensive_health_check


def main():
    parser = argparse.ArgumentParser(description="Receiver CLI")
    subparsers = parser.add_subparsers(dest='command', help='Available commands')

    # WebUI –∫–æ–º–∞–Ω–¥–∞
    webui_parser = subparsers.add_parser('webui', help='Start WebUI')
    webui_parser.add_argument('--port', type=int, default=7860, help='Port for WebUI')

    # Sync –∫–æ–º–∞–Ω–¥–∞
    sync_parser = subparsers.add_parser('sync', help='Sync protocols')
    sync_parser.add_argument('--date', type=str, help='Date to sync (YYYY-MM-DD)')
    sync_parser.add_argument('--days', type=int, default=1, help='Number of days to sync')
    sync_parser.add_argument('--limit', type=int, default=0, help='Limit number of protocols')

    # Download –∫–æ–º–∞–Ω–¥–∞
    download_parser = subparsers.add_parser('download', help='Download documents')
    download_parser.add_argument('--limit', type=int, default=0, help='Limit number of units')
    download_parser.add_argument('--dry-run', action='store_true', help='Dry run mode')

    # Health check –∫–æ–º–∞–Ω–¥–∞
    health_parser = subparsers.add_parser('health', help='Run health checks')

    args = parser.parse_args()

    if args.command == 'webui':
        from receiver.webui.app import main as webui_main
        webui_main()

    elif args.command == 'sync':
        service = EnhancedSyncService()
        if args.date:
            result = service.sync_date(args.date, limit=args.limit)
        else:
            result = service.sync_daily_updates(days=args.days, limit=args.limit)
        print(f"Sync completed: {result}")

    elif args.command == 'download':
        service = EnhancedProtocolDownloader()
        result = service.download_protocols(limit=args.limit, dry_run=args.dry_run)
        print(f"Download completed: {result}")

    elif args.command == 'health':
        results = run_comprehensive_health_check()
        for name, result in results.items():
            print(f"{name}: {result.status} - {result.message}")

    else:
        parser.print_help()


if __name__ == "__main__":
    main()
```
